## ğŸ“„ src/ResumeCreate/Education.js

```javascript
import React from "react";
import "./Resume.css";

class Education extends React.Component {
  render() {
    const education = [
      { id: 1, name: "Basaveshwar Science College", place: "Bagalkot", stddeg: "BCA", grade: "9.0" },
      { id: 2, name: "RV College of Engineering", place: "Bengaluru", stddeg: "MCA", grade: "9.1" }
    ];

    return (
      <div>
        <h2>Educational Qualifications</h2>
        <table>
          <thead>
            <tr>
              <th>ID</th>
              <th>Institution</th>
              <th>Place</th>
              <th>Degree</th>
              <th>CGPA</th>
            </tr>
          </thead>
          <tbody>
            {education.map(e => (
              <tr key={e.id}>
                <td>{e.id}</td>
                <td>{e.name}</td>
                <td>{e.place}</td>
                <td>{e.stddeg}</td>
                <td>{e.grade}</td>
              </tr>
            ))}
          </tbody>
        </table>
      </div>
    );
  }
}

export default Education;
```
