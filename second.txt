<!DOCTYPE html>
<html>
<head>
  <title>Student Information</title>
</head>
<body>
  <h2>Using GET Method</h2>
  <form action="/submit-get" method="get">
    <label>Name:</label><input type="text" name="name" required><br>
    <label>Branch:</label><input type="text" name="branch" required><br>
    <label>Semester:</label><input type="text" name="semester" required><br>
    <button type="submit">Submit (GET)</button>
  </form>

  <h2>Using POST Method</h2>
  <form action="/submit-post" method="post">
    <label>Name:</label><input type="text" name="name" required><br>
    <label>Branch:</label><input type="text" name="branch" required><br>
    <label>Semester:</label><input type="text" name="semester" required><br>
    <button type="submit">Submit (POST)</button>
  </form>
</body>
</html>






const express = require('express');
const path = require('path');
const app = express();
const port = 3000;

app.use(express.static(path.join(__dirname)));

app.use(express.urlencoded({ extended: true }));

app.get('/submit-get', (req, res) => {
  const { name, branch, semester } = req.query;
  const htmlResponse = `
    <h2>Student Information (GET)</h2>
    <p>Name: ${name}</p>
    <p>Branch: ${branch}</p>
    <p>Semester: ${semester}</p>
    <a href="/">Go Back</a>
  `;
  res.send(htmlResponse);
});

app.post('/submit-post', (req, res) => {
  const { name, branch, semester } = req.body;
  const htmlResponse = `
    <h2>Student Information (POST)</h2>
    <p>Name: ${name}</p>
    <p>Branch: ${branch}</p>
    <p>Semester: ${semester}</p>
    <a href="/">Go Back</a>
  `;
  res.send(htmlResponse);
});

app.listen(port, () => {
  console.log(`Server running at http://localhost:${port}`);
});
